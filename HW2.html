def visu_train_history(train_history,train_metric,validation_metric):
    plt.plot(train_history.history[train_metric])
    plt.plot(train_history.history[validation_metric])
    plt.title('Train History')
    plt.ylabel(train_metric)
    plt.xlabel('epoch')
    plt.legend(['train', 'validation'], loc='upper left')
    plt.show()

def plot_images_labels_prediction(images,    #
                                  labels,    #
                                  preds,     #
                                  index,    #inaey
                                  num = 5 ): #5
    fig = plt.gcf() # .Get Current Figure
    fig.set_size_inches(12, 6) #12.54 cm
    if num > 10:
        num = 10    # 10
    for i in range(0,num):
        ax = plt.subplot(2, 5, i + 1) # 
        
        ax.imshow (images[index])
        
        title = str(i) + ',' + label_dict[labels[index][0]] # title
        if len(preds) > 0:
            title += ' => ' + label_dict[preds[index]]
            
        ax.set_title(title,fontsize = 10)   # title
        index += 1
    plt.show()
